const KICKBOX_API_KEY = process.env.KICKBOX_API_KEY; // store in env / secrets

export async function verifyEmailWithKickbox(email) {
  const url = "https://api.kickbox.com/v2/verify";

  const res = await axios.get(url, {
    params: { email, apikey: KICKBOX_API_KEY, timeout: 6000 },
    timeout: 7000,
  });

  const body = res.data;

  if (!body.success) {
    // handle API failure gracefully
    throw new Error(`Kickbox verification failed: ${body.message || "Unknown"}`);
  }

  return {
    raw: body,
    mapped: {
      result: body.result,                // deliverable / undeliverable / risky / unknown
      reason: body.reason,
      is_role_email: body.role,
      is_free_email: body.free,
      is_disposable: body.disposable,
      is_catch_all: body.accept_all,
      email_quality_score: body.sendex,
      did_you_mean: body.did_you_mean,
    },
